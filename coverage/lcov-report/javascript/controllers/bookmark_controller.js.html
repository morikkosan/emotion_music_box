    this.onFrameLoad = (e) => {

    this.onChange = (e) => {

      else this.selected.delete(id)
      this.persist()
      this.fillHiddenFromStorage()
    }
    document.addEventListener("change", this.onChange)
  }
 
  disconnect() {
    document.removeEventListener("turbo:frame-load", this.onFrameLoad)
